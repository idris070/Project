365  void MeshLoader::ConstructSkeleton(KG::Skeleton_SmartPtr p_spSkeleton, const aiMesh * const p_pAiMesh, const aiNode * const p_AiNode)
366  {
367 
368  /*
369  To reconstruct the bone tree. Build a bone depth map by going through each Bone name and search them in the scene.
370  For each bone. calculate the depth from the root. the one with the lowest depth would be the root bone.
371  */
372 
373  //construct a bone depth map
374  std::map<std::string, unsigned> bone_depth_map; // bones must have different names.
375  for ( auto & bone_name : p_spSkeleton->names )
376  {
377  unsigned depth = 0; // depth starts from 1 at the root AiNode
378  const aiNode * scene_root_node = m_pScene->mRootNode;
379  if (!scene_root_node) // check root node first.
380  {
381  KE::Debug::print(KE::Debug::DBG_ERROR, "MeshLoader::ConstructSkeleton : root node is null?");
382  return;
383  }
384 
385  if (this->FindBoneDepth(depth, scene_root_node, bone_name))
386  {
387  bone_depth_map.insert(std::make_pair(bone_name, depth));
388  }
389  else
390  assert(false);
391  }
392 
393  // find root bones of skeleton.
394  // Note: there could be more than 1. (e.g. Blender's armture is a root but isn't technically a bone).
395  unsigned min_depth = 9999; // just something bigger than the possible number of bones...
396  std::string root_bone_name("N/A");
397  for (auto & it : bone_depth_map)
398  {
399  if (it.second < min_depth)
400  {
401  min_depth = it.second;
402  root_bone_name = it.first;
403  }
404  }
405  std::vector<std::string> root_bone_names;
406  for (auto & it : bone_depth_map) // get all root bone names.
407  {
408  if (it.second == min_depth)
409  root_bone_names.push_back(it.first);
410  }
411 
412  for (auto & name : root_bone_names)
413  {
414  this->GrowBoneTree(p_spSkeleton, nullptr, p_pAiMesh, this->FindAiNodeByName(name, p_AiNode));
