194  const glm::dquat offset = glm::angleAxis(delta, glm::dvec3(1.0, 0.0, 0.0));
195  glm::dquat Delta = camera_quat * offset * glm::conjugate(camera_quat);
196  Delta = Delta * node->GetOrientationQuat();
197  node->SetOrientationQuat(glm::normalize(Delta));
198  break;
199  }

204  KE::Debug::print("BEFORE: " + std::to_string(vec.x) + " " + std::to_string(vec.y) + " " + std::to_string(vec.z));

349  const bool LogicComponent::SpawnNewObject(const std::shared_ptr<CITS::NewObjectRequest_Event> p_spEvent)
350  {
351  /* figure out mesh and texture name. */

353  {

355  m_MeshName = "model55.x";
356  if (p_spEvent->GetTextureFileName() == "NONE")
357  m_TextureName = "NONE";
358  }
359  else if (p_spEvent->GetMeshFileName() == "DEFAULT")
360  {
361  KE::Debug::print("ObjectManipulator : Loading last imported model.");
362  }
363  else
364  {
365  m_MeshName = p_spEvent->GetMeshFileName();
366  m_TextureName = p_spEvent->GetTextureFileName();
367  }
368 
369  /* use texture? */
370  bool use_texture;
371  use_texture = (m_TextureName == "NONE") ? false : true; // don't use texture if string empty.
372 
373  std::string model_tex_dir = "models-textures/";
374  KG::MeshLoader loader;
375  auto mesh_list = loader.Load(model_tex_dir+m_MeshName)->GetList();
376  if (mesh_list.size() == 0)
377  {
378  KE::Debug::print("ObjectManipulator : nothing in mesh list.");
379  return false;
380  }
381 
382  KG::Mesh_SmartPtr mesh = mesh_list.at(0);
383  assert(mesh);
384  if(mesh->Loaded())
385  mesh->BufferAll();
386 
387  mesh->SetRenderPass(KG::RenderPass::Dynamic);
388  mesh->SetPosition(0.0,0.0,0.0);
389 
390  if (use_texture)
391  {
392  KG::Texture_SmartPtr texture(new KG::Texture(KG::Texture::DType::Tex2D, model_tex_dir+m_TextureName));
393  mesh->SetTexture(texture);
394  }
395 
396  // add entity component, and scenenode into their managers.
397  KG::Graphics::Get().GetScene().AddSceneNode(mesh);
398  KE::Entity_SmartPtr entity(new KE::Entity(mesh->GetEntityID()));
399  KE::RenderComponent_SmartPtr r_comp(new KE::RenderComponent(entity));
400  entity->AddComponent(r_comp);
401  r_comp->SetSceneNode(mesh);
402  KE::GetEntityManager().Add(entity);
403 
404  KE::Event::Get().QueueEvent(KE::Event_SmartPtr(new CITS::NewObject_Event(entity->GetID())));
405  return true;
